bool remove<T>() {
  final type = _getInjectType<T>();
  if (_singletonBinds.containsKey(type)) {
    var inject = _singletonBinds[type];
    _callDispose(inject);
    _singletonBinds.remove(type);
    return true;
  } else {
    return false;
  }
}
